def gaussseidel(A, b, x, e):

    def error(x1, x):
        return max(abs(x1 - x))

    n, m = A.shape

    Ad = A.diagonal().T
    bb = (b / Ad)    

    Aa = matrix([[(0 if i==j else (A[i, j]/A[i, i])) for j in range(m)]
                        for i in range(n)])

    xx = matrix(x, copy=True)    

    for counter in range(10):        
        x_prev = copy(xx)
        for i in range(n):
                        
            sum_a = sum(Aa[i, :i].getA1() * xx[:i].getA1())
            sum_b = sum(Aa[i, i+1:].getA1() * xx[i+1:].getA1())
            
            xn = bb[i] - (sum_a + sum_b);

            xx[i] = xn            
        if error(x_prev, xx) <= e:
            return xx        
    else:        
        raise Exception("  10 ,    .")

    return xx

from numpy import *
from functools import reduce
import operator

def lagrange(x_point, x, f):
    n = len(x)

    def b(i):
        lx = 1.0        
        for j in range(n):
            if j != i:
                lx *= (x_point - x[j]) / (x[i] - x[j])
        return lx

    L = [b(i) for i in range(len(x))]

    return sum(L[i] * f(x[i]) for i in range(n))

    # Test
