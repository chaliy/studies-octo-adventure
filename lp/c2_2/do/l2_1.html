def pareto(alternatives):
    def more_or_equal(x,y):
        return (x[0] >= y[0] and x[1] >= y[1])

    next = alternatives	
    for f in alternatives:	
        next = [x for x in next if (not (more_or_equal(f, x)) or x == f)]

    return next

def slater(alternatives):
    def more(x,y):
        return (x[0] > y[0] and x[1] > y[1])

    next = alternatives	
    for f in alternatives:	
        next = [x for x in next if (not (more(f, x)) or x == f)]		

    return next

import random
import unittest

import numpy
import numpy.testing as nt

from l2_1 import *

class TestL2_5(unittest.TestCase):      

    def test_pareto(self):
        
        source = [79, 95, 4, 37, 92, 95, 12, 52, 70, 14]
        alternatives = [[int(x/10),x%10] for x in source]
        results = pareto(alternatives)
     
        self.assertEqual(results, [[7, 9], [9, 5], [9, 5]])

    def test_slater(self):
        
        source = [79, 95, 4, 37, 92, 95, 12, 52, 70, 14]
        alternatives = [[int(x/10),x%10] for x in source]
        results = slater(alternatives)
     
        self.assertEqual(results, [[7, 9], [9, 5], [9, 2], [9, 5]])

def plot_pareto_slater(source):
    def plot_result(all_items, selected, name):
        scatter([i[0] for i in all_items],
                [i[1] for i in all_items])

        plot([i[0] for i in selected],
             [i[1] for i in selected])
        
        title(name)
        
    alternatives = [[int(x/10),x%10] for x in source]
    print("Source: ")    
    display_table([], transpose([["Q1", "Q2"]] + alternatives))

    pareto_results = pareto(alternatives)
    print("Pareto: ")
    display_table([], transpose([["Q1", "Q2"]] + pareto_results))
    subplot(1, 2, 1)
    plot_result(alternatives, pareto_results, "Pareto")

    slater_results = slater(alternatives)
    print("Slater: ")
    display_table([], transpose([["Q1", "Q2"]] + slater_results))
    subplot(1, 2, 2)
    plot_result(alternatives, slater_results, "Slater")
